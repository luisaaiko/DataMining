price_metrics <- function(pred_log, actual_price) {
  pred_price <- exp(as.numeric(pred_log))
  mse <- mean((actual_price - pred_price)^2)
  rmse <- sqrt(mse)
  mae <- mean(abs(actual_price - pred_price))
  r2 <- 1 - sum((actual_price - pred_price)^2) /
    sum((actual_price - mean(actual_price))^2)
  
  tibble(
    RMSE = rmse,
    MAE = mae,
    MSE = mse,
    R2 = r2
  )
}

print_price_metrics <- function(model_name, pred_log, actual_price) {
  out <- price_metrics(pred_log, actual_price)
  cat(model_name, "\n")
  cat("RMSE:", round(out$RMSE, 3), "\n")
  cat("MAE :", round(out$MAE, 3), "\n")
  cat("MSE :", round(out$MSE, 3), "\n")
  cat("R^2 :", round(out$R2, 6), "\n\n")
  invisible(out)
}

extract_oob_mse <- function(rf_fit) {
  rf_fit$prediction.error
}

extract_oob_rsq <- function(rf_fit) {
  rf_fit$r.squared

11 Boosting

Boosting builds trees sequentially, with each new tree focusing on errors made by the previous ones. Here we tune interaction depth, shrinkage, and minimum node size, then choose the best number of trees by cross-validation for each combination.

set.seed(5703)

boost_grid <- expand.grid(
  interaction.depth = c(1, 3, 5),
  shrinkage = c(0.005, 0.01),
  n.minobsinnode = c(5, 10),
  KEEP.OUT.ATTRS = FALSE
) %>%
  as_tibble()

boost_tuning_results <- pmap_dfr(boost_grid, function(interaction.depth, shrinkage, n.minobsinnode) {
  fit <- gbm(
    log_price ~ . - price,
    data = train_df,
    distribution = "gaussian",
    n.trees = 4000,
    interaction.depth = interaction.depth,
    shrinkage = shrinkage,
    n.minobsinnode = n.minobsinnode,
    cv.folds = 5,
    n.cores = N_CORES,
    bag.fraction = 0.5,
    verbose = FALSE
  )
  
  best_iter <- gbm.perf(fit, method = "cv", plot.it = FALSE)
  best_cv_error <- fit$cv.error[best_iter]
  
  tibble(
    interaction.depth = interaction.depth,
    shrinkage = shrinkage,
    n.minobsinnode = n.minobsinnode,
    best_iter = best_iter,
    CV_Error = best_cv_error
  )
})

  arrange(CV_Error) %>%

14 Interpretation of Results

Based on the comparison table, the best model should be identified primarily by the lowest RMSE and MAE, along with the highest R^2 on the test set. In general: The decision tree serves as a simple benchmark and is usually less accurate than the ensemble methods. Bagging often improves over a single tree by reducing variance. Random forest often improves further by decorrelating trees through random predictor selection. Boosting can perform very well because it builds trees sequentially to correct earlier errors. The final recommendation should be based on the model that performs best on the held-out test set while remaining interpretable and appropriately tuned.
